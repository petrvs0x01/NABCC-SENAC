Var

   delta,a,b,c : real
   raiz1,raiz2 : real

      escreval("Escreva o valor dos coeficientes")

         delta <- Raizq((b*b) - 4*a*c))
         se delta < 0 entao
              escreval("Não é possível efetuar o cálculo")
         senao
              raiz1 <- ((-b + delta) / (2*a))
              raiz2 <- ((-b - delta) / (2*a))
              escreval("X+ = ",raiz1:1:4)
              escreval("X- = ",raiz2:1:4)
         fimse
      senao
